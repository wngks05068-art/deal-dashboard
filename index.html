<script>

// Stage labels

const STAGE_LABELS = {

  s1: 'S1 - 발굴',

  s2: 'S2 - 검토',

  s3: 'S3 - 실사',

  s4: 'S4 - 협상',

  s5: 'S5 - 완료'

};


// Initialize data

let dealData = {

  companies: [

    {

      id: "c1",

      name: "(주)아이메디",

      sector: "소비재",

      amount: 30,

      history: [{ stage: "s3", date: "2026-04-28" }],

      plannedNext: { stage: "s5", date: "2026-04-30" },

      note: ""

    },

    {

      id: "c2",

      name: "투디엠",

      sector: "소재",

      amount: 20,

      history: [{ stage: "s3", date: "2026-05-07" }],

      plannedNext: { stage: "s5", date: "2026-05-15" },

      note: "납입 5월 중순(추정)"

    },

    {

      id: "c3",

      name: "어썸레이",

      sector: "반도체",

      amount: 10,

      history: [{ stage: "s3", date: "2026-05-07" }],

      plannedNext: { stage: "s5", date: null },

      note: "납입일 미정"

    },

    {

      id: "c4",

      name: "코스모웍스",

      sector: "미분류",

      amount: 20,

      history: [{ stage: "s3", date: "2026-05-07" }],

      plannedNext: { stage: "s5", date: "2026-05-11" },

      note: ""

    },

    {

      id: "c5",

      name: "데브크라",

      sector: "AI",

      amount: 20,

      history: [{ stage: "s3", date: "2026-05-07" }],

      plannedNext: { stage: "s5", date: null },

      note: "납입일 미정"

    }

  ]

};


// Load from localStorage if exists

const saved = localStorage.getItem('dealData');

if (saved) {

  try { dealData = JSON.parse(saved); } catch(e) {}

}


function saveToStorage() {

  localStorage.setItem('dealData', JSON.stringify(dealData));

}


function generateId() {

  return 'c' + Date.now();

}


function getLatestHistory(company) {

  if (!company.history || company.history.length === 0) return { stage: '-', date: '-' };

  return company.history[company.history.length - 1];

}


function renderTable() {

  const tbody = document.getElementById('dealTableBody');

  const empty = document.getElementById('emptyState');

  const companies = dealData.companies;

  

  document.getElementById('totalCount').textContent = companies.length;

  document.getElementById('totalAmount').textContent = companies.reduce((sum, c) => sum + (c.amount || 0), 0);

  

  if (companies.length === 0) {

    tbody.innerHTML = '';

    empty.style.display = 'block';

    return;

  }

  empty.style.display = 'none';

  

  tbody.innerHTML = companies.map((c, i) => {

    const latest = getLatestHistory(c);

    const nextStage = c.plannedNext?.stage || '-';

    const nextDate = c.plannedNext?.date || '미정';

    

    return `<tr>

      <td>${i + 1}</td>

      <td><strong>${c.name}</strong></td>

      <td>${c.sector || '-'}</td>

      <td class="amount-text">${c.amount}억</td>

      <td><span class="stage-badge stage-${latest.stage}">${STAGE_LABELS[latest.stage] || latest.stage}</span></td>

      <td>${latest.date || '-'}</td>

      <td><span class="stage-badge stage-${nextStage}">${STAGE_LABELS[nextStage] || nextStage}</span></td>

      <td>${nextDate}</td>

      <td class="note-text">${c.note || '-'}</td>

      <td><button class="btn btn-secondary" onclick="showHistory('${c.id}')">이력(${c.history?.length || 0})</button></td>

      <td>

        <div style="display:flex;gap:4px;">

          <button class="btn btn-edit" onclick="openEditModal('${c.id}')">수정</button>

          <button class="btn btn-danger" onclick="deleteDeal('${c.id}')">삭제</button>

        </div>

      </td>

    </tr>`;

  }).join('');

}


// Modal controls

function openAddModal() {

  document.getElementById('modalTitle').textContent = '딜 추가';

  document.getElementById('editId').value = '';

  document.getElementById('inputName').value = '';

  document.getElementById('inputSector').value = '미분류';

  document.getElementById('inputAmount').value = '';

  document.getElementById('inputStage').value = 's3';

  document.getElementById('inputDate').value = '';

  document.getElementById('inputNextStage').value = 's5';

  document.getElementById('inputNextDate').value = '';

  document.getElementById('inputNote').value = '';

  document.getElementById('dealModal').classList.add('active');

}


function openEditModal(id) {

  const c = dealData.companies.find(x => x.id === id);

  if (!c) return;

  

  const latest = getLatestHistory(c);

  

  document.getElementById('modalTitle').textContent = '딜 수정';

  document.getElementById('editId').value = id;

  document.getElementById('inputName').value = c.name;

  document.getElementById('inputSector').value = c.sector || '미분류';

  document.getElementById('inputAmount').value = c.amount;

  document.getElementById('inputStage').value = latest.stage || 's3';

  document.getElementById('inputDate').value = latest.date || '';

  document.getElementById('inputNextStage').value = c.plannedNext?.stage || '';

  document.getElementById('inputNextDate').value = c.plannedNext?.date || '';

  document.getElementById('inputNote').value = c.note || '';

  document.getElementById('dealModal').classList.add('active');

}


function closeModal() {

  document.getElementById('dealModal').classList.remove('active');

}


function saveDeal() {

  const id = document.getElementById('editId').value;

  const name = document.getElementById('inputName').value.trim();

  const sector = document.getElementById('inputSector').value;

  const amount = parseInt(document.getElementById('inputAmount').value) || 0;

  const stage = document.getElementById('inputStage').value;

  const date = document.getElementById('inputDate').value || null;

  const nextStage = document.getElementById('inputNextStage').value || null;

  const nextDate = document.getElementById('inputNextDate').value || null;

  const note = document.getElementById('inputNote').value.trim();

  

  if (!name) { alert('회사명을 입력해 주세요.'); return; }

  if (!amount) { alert('금액을 입력해 주세요.'); return; }

  

  if (id) {

    // Edit

    const c = dealData.companies.find(x => x.id === id);

    if (!c) return;

    c.name = name;

    c.sector = sector;

    c.amount = amount;

    c.note = note;

    

    const latest = getLatestHistory(c);

    if (latest.stage !== stage || latest.date !== date) {

      c.history.push({ stage: stage, date: date });

    }

    c.plannedNext = { stage: nextStage, date: nextDate };

  } else {

    // Add

    dealData.companies.push({

      id: generateId(),

      name: name,

      sector: sector,

      amount: amount,

      history: [{ stage: stage, date: date }],

      plannedNext: { stage: nextStage, date: nextDate },

      note: note

    });

  }

  

  saveToStorage();

  renderTable();

  closeModal();

}


function deleteDeal(id) {

  const c = dealData.companies.find(x => x.id === id);

  if (!c) return;

  if (!confirm(`"${c.name}" 딜을 삭제하시겠습니까?`)) return;

  dealData.companies = dealData.companies.filter(x => x.id !== id);

  saveToStorage();

  renderTable();

}


// History modal

function showHistory(id) {

  const c = dealData.companies.find(x => x.id === id);

  if (!c) return;

  document.getElementById('historyTitle').textContent = `${c.name} 진행 이력`;

  const list = document.getElementById('historyList');

  

  if (!c.history || c.history.length === 0) {

    list.innerHTML = '<li>이력이 없습니다.</li>';

  } else {

    list.innerHTML = c.history.map((h, i) => 

      `<li><strong>${STAGE_LABELS[h.stage] || h.stage}</strong> — ${h.date || '날짜 미정'}${i === c.history.length - 1 ? ' (현재)' : ''}</li>`

    ).join('');

  }

  document.getElementById('historyModal').classList.add('active');

}


function closeHistoryModal() {

  document.getElementById('historyModal').classList.remove('active');

}


// JSON Export

function exportJSON() {

  const blob = new Blob([JSON.stringify(dealData, null, 2)], { type: 'application/json' });

  const url = URL.createObjectURL(blob);

  const a = document.createElement('a');

  a.href = url;

  a.download = 'deal_data_' + new Date().toISOString().slice(0,10) + '.json';

  a.click();

  URL.revokeObjectURL(url);

}


// JSON Import

function importJSON() {

  document.getElementById('fileInput').click();

}


function handleFileImport(event) {

  const file = event.target.files[0];

  if (!file) return;

  const reader = new FileReader();

  reader.onload = function(e) {

    try {

      const imported = JSON.parse(e.target.result);

      if (imported.companies && Array.isArray(imported.companies)) {

        dealData = imported;

        saveToStorage();

        renderTable();

        alert('JSON 파일을 성공적으로 불러왔습니다!');

      } else {

        alert('올바른 딜 데이터 형식이 아닙니다.');

      }

    } catch(err) {

      alert('JSON 파일 파싱 오류: ' + err.message);

    }

  };

  reader.readAsText(file);

  event.target.value = '';

}

renderTable();
